## CONVERT

```sql
CONVERT (value, type)

-- type
TIME 	 -- 'HH:MM:SS'.
DATE 	 -- 'YYYY-MM-DD'
DATETIME -- 'YYYY-MM-DD HH:MM:SS'
CHAR	 -- строка фиксированной длины
SIGNED	 -- целое число со знаком
UNSIGNED -- целое число без знака
BINARY	 -- двоичная строка

-- ПРИМЕР
SELECT CONVERT('2016-03-07', DATE); -- 2016-03-07
SELECT CONVERT('08:36:57', TIME);	-- 08:36:57
```

## STR_TO_DATE

```sql
STR_TO_DATE (string, format_mask)
-- Функция возвращает
DATETIME -- если строка содержит дату и временя
DATE	 -- если строка содержит дату
TIME	 -- если строка содержит временя
NULL	 -- если не может извлечь дату и временя (используя format_mask)

-- ПРИМЕР
SELECT
STR_TO_DATE('February 13 2015', '%M %d %Y'); -- 2015-02-13 
STR_TO_DATE('March,14,2015', '%M,%e,%Y'); 	 -- 2015-03-14 
STR_TO_DATE('10,15,30', '%h,%i,%s'); 		 -- 10:15:30
STR_TO_DATE('2014,2,28 09', '%Y,%m,%d %h');  -- 2014-02-28 09:00:00 
STR_TO_DATE('2014,2,28 09,30,05', '%Y,%m,%d %h,%i,%s'); -- 2014-02-28 09:30:05 
STR_TO_DATE('Friday, February 13, 2015', '%W, %M %e, %Y'); -- 2015-02-13 
```

### Формат

```sql
-- ГОД
%Y	-- в виде числового значения из 4 цифр
%y	-- в виде числового значения из 2 цифр

-- МЕСЯЦ
%c	-- Числовое значение  (0 до 12)
%b	-- Сокращенное название месяца (Jan до Dec)
%M	-- Название полностью (January to December)
%m	-- Название в виде числового значения (от 00 до 12)

-- НЕДЕЛЯ
%W	-- Полностью (Sunday to Saturday)
%a	-- Сокращенное (Sun до Sat)
%w	-- Sunday=0 and Saturday=6
%U	-- Sunday - первый день недели (от 00 до 53)
%u	-- Monday - это первый день недели (от 00 до 53)
%V	-- Sunday - первый день недели (от 01 до 53). Доступно начиная с версии MySQL 3.23.8 и используется с %X
%v	-- Monday - первый день недели (от 01 до 53). Доступно начиная с версии MySQL 3.23.8 и используется с %X

-- ДЕНЬ
%d	-- значение 01 - 31
%e	-- значение 0 - 31
%j	-- значение 001 - 366
%D	-- значение, за которым следует суффикс (1st, 2nd, 3rd, ...)

-- ЧАС
%H	-- от 00 до 23
%h	-- от 00 до 12
%I	-- от 00 до 12
%k	-- от 00 до 23
%l	-- от 1 до 12

-- МИНУТЫ
%i	-- от 00 до 59

-- СЕКУНДЫ
%S	-- от 00 до 59
%s	-- от 00 до 59

-- МИКРОСЕКУНДЫ
%f	-- от 000000 до 999999 (доступны начиная с MySQL 4.1.1)

-- ВРЕМЯ
%p	-- До или после полудня AM или PM
%r	-- Время в 12-часовом формате AM или PM (hh:mm:ss AM/PM)
%T	-- Время в 24-часовом формате (hh:mm:ss)

-- ПРОЧЕЕ
%X	-- Год недели, где Sunday - первый день недели. Доступно начиная с версии MySQL 3.23.8 и используется с% V
%x	-- Год недели, где Monday - первый день недели. Доступно начиная с MySQL 3.23.8 и используется с %v

```
