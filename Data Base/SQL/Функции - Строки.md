несколько встроенных функций для работы со строками:

```sql
CONCAT('text', ' ', 'num') 	  -- объединение строк



LENGTH 	  -- возвращает количество символов в строке
TRIM 	  -- удаляет пробелы в начале и конце строки
SUBSTRING -- извлекает подстроку из строки
REPLACE   -- заменяет подстроку в строке
LOWER 	  -- переводит символы строки в нижний регистр
UPPER 	  -- переводит символы строки в верхний регистр и т.д.
```

---

```sql
SUBSTRING(		-- извление символов
	value, 		-- обрабатываемый текст
	offset, 	-- смещение
	count_simb	-- колличество символов от начала
) 

-- Пример
SELECT 
	id, 
	name, 
	SUBSTRING(birthday_at, 1, 3) AS decade -- 198, 199, 200, 201(0) 
FROM users 
ORDER BY decade; -- соритровка по десятилетиям
```

```sql
CONCAT(value, [value...]) -- соединение строк
```
